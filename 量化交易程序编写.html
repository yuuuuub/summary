import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler, MinMaxScaler
import matplotlib.pyplot as plt

def clean_financial_data(df):
    """
    金融数据清洗主函数
    """
    print(f"原始数据形状: {df.shape}")
    
    # 1. 处理缺失值
    print(f"缺失值统计:\n{df.isnull().sum()}")
    
    # 前向填充处理缺失值
    df = df.ffill()
    
    # 2. 异常值检测与处理
    returns = df['Close'].pct_change()
    volatility = returns.rolling(20).std()
    
    # 过滤极端收益率(3倍标准差)
    mask = np.abs(returns) < 3 * volatility
    df = df[mask]
    
    # 3. 数据标准化
    scaler = StandardScaler()
    df[['Close_norm', 'Volume_norm']] = scaler.fit_transform(
        df[['Close', 'Volume']]
    )
    
    print(f"清洗后数据形状: {df.shape}")
    return df

技术指标计算与实现

移动平均线 (MA)

简单移动平均(SMA)、指数移动平均(EMA)、加权移动平均(WMA) 用于识别趋势方向和支撑阻力位。

相对强弱指数 (RSI)

衡量价格变动速度和幅度的动量振荡器， RSI > 70为超买，RSI < 30为超卖。

MACD指标

移动平均收敛发散指标，通过快慢均线差值 判断买卖时机和趋势强度。

布林带 (Bollinger Bands)

基于移动平均线和标准差构建的通道指标， 用于判断价格波动范围和突破信号。

技术指标完整实现 

import talib
import pandas as pd
import numpy as np

class TechnicalIndicators:
    """技术指标计算类"""
    
    def __init__(self, data):
        self.data = data
        self.close = data['Close'].values
        self.high = data['High'].values
        self.low = data['Low'].values
        self.volume = data['Volume'].values
    
    def calculate_ma(self, period=20, ma_type='SMA'):
        """计算移动平均线"""
        if ma_type == 'SMA':
            return self.data['Close'].rolling(window=period).mean()
        elif ma_type == 'EMA':
            return talib.EMA(self.close, timeperiod=period)
        elif ma_type == 'WMA':
            return talib.WMA(self.close, timeperiod=period)
    
    def calculate_rsi(self, period=14):
        """计算RSI指标"""
        return talib.RSI(self.close, timeperiod=period)
    
    def calculate_macd(self, fast=12, slow=26, signal=9):
        """计算MACD指标"""
        macd, macd_signal, macd_hist = talib.MACD(
            self.close, 
            fastperiod=fast, 
            slowperiod=slow, 
            signalperiod=signal
        )
        return macd, macd_signal, macd_hist
    
    def calculate_bollinger_bands(self, period=20, std_dev=2):
        """计算布林带"""
        upper, middle, lower = talib.BBANDS(
            self.close, 
            timeperiod=period, 
            nbdevup=std_dev, 
            nbdevdn=std_dev
        )
        return upper, middle, lower
    
    def calculate_all_indicators(self):
        """计算所有技术指标"""
        result = self.data.copy()
        
        # 移动平均线
        result['SMA_20'] = self.calculate_ma(20, 'SMA')
        result['EMA_12'] = self.calculate_ma(12, 'EMA')
        
        # RSI
        result['RSI_14'] = self.calculate_rsi(14)
        
        # MACD
        macd, macd_signal, macd_hist = self.calculate_macd()
        result['MACD'] = macd
        result['MACD_Signal'] = macd_signal
        result['MACD_Hist'] = macd_hist
        
        # 布林带
        bb_upper, bb_middle, bb_lower = self.calculate_bollinger_bands()
        result['BB_Upper'] = bb_upper
        result['BB_Middle'] = bb_middle
        result['BB_Lower'] = bb_lower
        
        return result

# 使用示例
data = yf.download("AAPL", start="2023-01-01", end="2024-01-01")

result = indicators.calculate_all_indicators()

# 显示结果

主流交易策略实现

1. 均线交叉策略（Moving Average Crossover）

策略原理

基于短期与长期均线的交互关系判断市场趋势：

  • 金叉信号：短期均线从下方突破长期均线时买入
  • 死叉信号：长期均线反向突破短期均线时卖出
Python实现示例
import backtrader as bt

class SmaCross(bt.Strategy):
    params = (('pfast', 5), ('pslow', 20))
    
    def __init__(self):
        sma1 = bt.ind.SMA(period=self.p.pfast)
        sma2 = bt.ind.SMA(period=self.p.pslow)
        self.crossover = bt.ind.CrossOver(sma1, sma2)
    
    def next(self):
        if not self.position:
            if self.crossover > 0: 
                self.buy()  # 金叉买入
        elif self.crossover < 0: 
            self.close()  # 死叉卖出

2. 动量策略（Momentum Strategy）

策略原理

基于"趋势延续性"假设，捕捉资产价格持续上涨或下跌的趋势[223][224]：

  • 近期表现优异的资产倾向于继续保持强势
  • 表现疲软的资产可能持续弱势

3. 均值回归策略（Mean Reversion Strategy）

策略原理

基于价格波动的统计特性，认为资产价格偏离历史均值后将大概率回归[215][217][219]：

  • 价格高于均值时卖出，低于均值时买入
  • 使用Z-score衡量价格偏离程度
Z-score均值回归实现
# Z-score均值回归策略
rolling_mean = data['Close'].rolling(window=20).mean()
rolling_std = data['Price'].rolling(window=30).std()
z_score = (data['Price'] - rolling_mean) / rolling_std

# 生成交易信号
buy_signal = z_score < -1.0   # Z-score < -1时买入
sell_signal = z_score > 1.0   # Z-score > 1时卖出

# 持仓与资金管理
position = 0
capital = 10000
for i in range(len(data)):
    if buy_signal[i] and position == 0:
        position = capital // data['Price'][i]
        capital = 0
    elif sell_signal[i] and position > 0:

        position = 0

class SignalGenerator:
    def __init__(self, strategy_config):
        self.config = strategy_config
        self.signals = {}
    
    def generate_ma_signal(self, data, short_period=5, long_period=20):
        """均线交叉信号"""
        short_ma = data['close'].rolling(short_period).mean()
        long_ma = data['close'].rolling(long_period).mean()
        return np.where(short_ma > long_ma, 1, -1)
    
    def generate_momentum_signal(self, data, lookback=12):
        """动量信号"""
        returns = data['close'].pct_change(lookback)
        return np.sign(returns)
    
    def generate_mean_reversion_signal(self, data, window=20, threshold=1.5):
        """均值回归信号"""
        rolling_mean = data['close'].rolling(window).mean()
        rolling_std = data['close'].rolling(window).std()
        z_score = (data['close'] - rolling_mean) / rolling_std
        return np.where(z_score < -threshold, 1, 
                       np.where(z_score > threshold, -1, 0))
    
    def combine_signals(self, signals, weights=None):
        """组合多个信号"""
        if weights is None:
            weights = [1.0] * len(signals)
        
        combined = np.zeros_like(signals[0])
        for signal, weight in zip(signals, weights):
            combined += signal * weight
        
        return np.sign(combined)

性能指标计算

核心指标公式[1]

年化收益率

衡量策略的年化收益能力

年化收益率 = (最终价值/初始价值)^(1/年数) - 1
夏普比率

衡量风险调整后的收益

夏普比率 = (年化收益率 - 无风险利率)/年化收益率标准差
最大回撤

衡量策略的最大风险

最大回撤 = (峰值 - 谷值)/峰值

import numpy as np

def annual_return(returns):
    """计算年化收益率"""
    return np.prod(1 + returns) ** (252/len(returns)) - 1

def sharpe_ratio(returns, risk_free=0):
    """计算夏普比率"""
    excess = returns - risk_free
    return np.mean(excess) / np.std(excess) * np.sqrt(252)

def max_drawdown(returns):
    """计算最大回撤"""
    cum = (1 + returns).cumprod()
    peak = cum.expanding().max()
    return (peak - cum).max() / peak.max()

class PerformanceAnalyzer:
    """性能分析器"""
    
    def __init__(self, returns):
        self.returns = returns
        
    def get_metrics(self):
        """获取所有性能指标"""
        return {
            'annual_return': annual_return(self.returns),
            'sharpe_ratio': sharpe_ratio(self.returns),
            'max_drawdown': max_drawdown(self.returns),
            'volatility': np.std(self.returns) * np.sqrt(252),
            'win_rate': len(self.returns[self.returns > 0]) / len(self.returns)

    def grid_search(self, param_grid, metric='sharpe_ratio'):
        """网格搜索优化"""
        param_combinations = list(itertools.product(*param_grid.values()))
        param_names = list(param_grid.keys())
        
        # 并行执行回测
        with ProcessPoolExecutor() as executor:
            futures = []
            for params in param_combinations:
                param_dict = dict(zip(param_names, params))
                future = executor.submit(self._single_backtest, param_dict)
                futures.append((param_dict, future))

        for param_dict, future in futures:
            try:
                result = future.result()
                result['params'] = param_dict
                self.results.append(result)
            except Exception as e:
                print(f"参数组合 {param_dict} 执行失败: {e}")
                
        # 按指标排序
        self.results.sort(key=lambda x: x.get(metric, 0), reverse=True)
        return self.results

  • 波动率适配：ATR值高时减少单笔头寸，ATR值低时适度加仓
  • 资金保护：设定单ATR波动对应总资金的固定比例（如1%）
  • 动态调整：根据市场波动性实时重新计算头寸规模

头寸计算公式

买入数量 × ATR = 总资金 × 1%
示例：总资金10万元，ATR=0.201元，则头寸规模 = 1000元 ÷ 0.201 ≈ 4900股[60]

# ATR动态仓位管理实现
import pandas as pd
import numpy as np

class PositionManager:
    def __init__(self, capital, atr_window=14):
        self.capital = capital
        self.atr_window = atr_window
        self.position = 0
    
    def calculate_atr(self, data):
        high = data['High']
        low = data['Low']
        close = data['Close']
        tr = pd.concat([
            high - low,
            abs(high - close.shift(1)),
            abs(low - close.shift(1))
        ], axis=1).max(axis=1)
        return tr.rolling(self.atr_window).mean()
    
    def dynamic_position(self, data):
        atr_value = self.calculate_atr(data)
        # 计算目标头寸规模
        target_pos = int(self.capital * 0.01 / 
                        atr_value.iloc[-1])
        
        # 加减仓逻辑
        if abs(target_pos - self.position) > 100:
            self.position = target_pos
            return target_pos
        return self.position

def risk_parity_allocation(returns, capital):
    """
    基于风险平价的资金分配
    """
    # 计算协方差矩阵
    cov_matrix = returns.cov()
    
    # 计算各资产波动率
    volatility = np.sqrt(np.diag(cov_matrix))
    
    # 风险平价权重（与波动率成反比）
    weights = 1 / volatility
    weights = weights / weights.sum()
    
    # 分配资金
    allocation = capital * weights
    
    return allocation

class DualMAStrategy:
    """双均线交叉策略"""
    
    def __init__(self, config: Dict):
        self.short_window = config['short_window']  # 10
        self.long_window = config['long_window']    # 30
        self.threshold = config.get('threshold', 0.02)  # 2%阈值
        
    def generate_signals(self, data: pd.DataFrame) -> List[Dict]:
        """生成交易信号"""
        signals = []
        
        for symbol in data.columns.levels[1]:
            symbol_data = data.xs(symbol, level=1, axis=1)
            
            # 计算移动平均线
            short_ma = symbol_data['Close'].rolling(self.short_window).mean()
            long_ma = symbol_data['Close'].rolling(self.long_window).mean()
            
            # 生成信号
            current_short = short_ma.iloc[-1]
            current_long = long_ma.iloc[-1]
            prev_short = short_ma.iloc[-2]
            prev_long = long_ma.iloc[-2]
            
            # 金叉买入信号
            if (prev_short <= prev_long and current_short > current_long and
                (current_short - current_long) / current_long > self.threshold):
                signals.append({
                    'symbol': symbol,
                    'action': 'buy',
                    'price': symbol_data['Close'].iloc[-1],
                    'timestamp': symbol_data.index[-1],
                    'confidence': min((current_short - current_long) / current_long, 0.1)
                })
            
            # 死叉卖出信号
            elif (prev_short >= prev_long and current_short < current_long and
                  (current_long - current_short) / current_long > self.threshold):
                signals.append({
                    'symbol': symbol,
                    'action': 'sell',
                    'price': symbol_data['Close'].iloc[-1],
                    'timestamp': symbol_data.index[-1],
                    'confidence': min((current_long - current_short) / current_long, 0.1)
                })
                
        return signals

随着金融科技的快速发展，量化交易系统需要在多个维度持续优化以保持竞争优势。根据2025年技术发展趋势[144][146][148]，优化方向主要集中在以下几个核心领域：

AI智能化

机器学习模型融合，自动化策略生成

执行优化

低延迟系统设计，硬件加速

风险增强

多层防御架构，实时监控预警

机器学习应用扩展

机器学习已成为量化交易的重要工具，通过从海量数据中提取信息、预测市场趋势、优化交易策略，显著提升了量化交易的效率与精准度[7][8][9]。

预测建模

  • LSTM网络：处理时间序列数据，捕捉长期依赖关系[7][8]
  • Transformer模型：处理多模态数据，提升预测精度[11]
  • 强化学习：动态优化策略参数，适应市场变化[11]

信号挖掘

  • XGBoost算法：提取非线性Alpha信号[8][11]
  • 聚类分析：识别市场异常模式[11]
  • NLP技术：分析新闻舆情，捕捉市场情绪[7][9]

实施建议：

从简单的线性模型开始，逐步引入深度学习模型。重点关注模型可解释性，满足金融监管需求[7]。建议采用AutoML工具缩短策略开发周期[144][148]。

执行速度优化方案

量化交易系统的执行速度直接影响交易效率和盈利能力，尤其在高频交易场景中，毫秒级延迟可能带来显著收益差异[45][47]。

构建多层防御架构，实现"预防-监控-处置"三位一体的风险控制体系[13][15][17]。
